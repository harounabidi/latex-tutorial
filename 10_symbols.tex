\documentclass{article}
  
\begin{document}

  % Remember that everything after % will be treated as a comment!!
  
  Maybe I have now learnt about 1% of \LaTeX.

  Maybe I have now learnt about 1\% of \LaTeX.
  
  ---

  % Use double backslash \\ to start a new line

  This is the first line.\\ This is the second line

  ---

  % We can use a parameter after \\ to specify the distance between the lines

  This is the first line.\\[1cm] This is the second line
  
\end{document}
